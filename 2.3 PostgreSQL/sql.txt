CREATE TABLE employees
(
    id SERIAL PRIMARY KEY,
    firstname VARCHAR(30),
    lastname VARCHAR(30),
    dateofbirth DATE,
    start_work DATE,
    job_title VARCHAR,
    employee_level CHAR(6),
    salary MONEY,
    department_id INT,
    drivers_license BOOLEAN    
)
